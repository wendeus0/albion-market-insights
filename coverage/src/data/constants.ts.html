function genIds(types: string[]): string[] {
  return ['T4', 'T5', 'T6', 'T7', 'T8'].flatMap(tier =>
    types.flatMap(type =>
      ENCHANTMENT_LEVELS.map(level =>
        level === 0 ? `${tier}_${type}` : `${tier}_${type}@${level}`
      )
    )
  );
}
